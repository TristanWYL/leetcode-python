# Solution

```python
class Solution:
    def countAndSay(self, n: int) -> str:
        if n == 1: return "1"
        pre = self.countAndSay(n - 1)
        cur = pre[0]
        count = 0
        result = ""
        for c in pre:
            if c == cur:
                count += 1
            else:
                result = f"{result}{count}{cur}"
                cur = c
                count = 1
        result = f"{result}{count}{cur}"
        return result
```
